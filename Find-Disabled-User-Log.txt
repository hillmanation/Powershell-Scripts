$csvfolder = "\_SDS Ops Folder\Compass Logs\AD Account Disable Logs\2020"

$monthfolders = (Get-ChildItem -Path $csvfolder -Directory).Name
$output = @()

ForEach ($monthfolder in $monthfolders) {

    $filename = "$csvfolder\$monthfolder\users\AD_User_Disabled_Audit_*.csv"
    
    $disabledusers = Import-CSV $filename

    ForEach ($disableduser in $disabledusers) {
        
        If ($disableduser.Name -eq 'Roberts, Brian') { $output += $disableduser }
    }
}

$output | FT -AutoSize
