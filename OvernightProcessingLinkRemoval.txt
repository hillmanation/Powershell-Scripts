$computers = (Get-ADComputer -filter * -properties Name,OperatingSystem | where { $_.OperatingSystem -like "*windows*" -and $_.Name -notlike "*GOLD*" } | Select Name).Name

ForEach ($computer in $computers) {

    $file = "\\$computer\c$\Users\Public\Desktop\Overnight Processing Request.lnk"

    If ((Test-Path -Path "\\$computer\c$\Users\Public\Desktop\Overnight Processing Request.lnk" -ErrorAction SilentlyContinue)) { 
    
        Remove-Item -Path "\\$computer\c$\Users\Public\Desktop\Overnight Processing Request.lnk" -Force -Confirm:$false
        echo "Link removed from $computer..."
        "Link removed from $computer..." | Out-File -FilePath "\\aetpdata\ourstuff\_Hillman\Script Output\OvernightProcessLinkRemoved.txt"
    }
    Else { echo "Path not found on $computer, noting so it can be checked..."; $nolink += $computer}

}
